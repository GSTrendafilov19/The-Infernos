77// class GRand
78// Good random number generation
79// Convenience wrapper around C++ Standard Library RNG.
80// Exceptions:
81// Public member functions that take arguments of arbitrary types
82// (ctor from seed, seed [one parameter], operator() [one
83// parameter]) will throw what & when an operation on the argument
84// type throws. No such exceptions will be thrown if the argument
85// is of a fundamental type.
86// If the GRand object has been seeded unpredictably (default ctor,
87// seed [no parameters]), then public member functions that
88// generate random values (i, d, b, operator()) may throw the
89// exception thrown by std::random_device on failure. No such
90// exceptions will be thrown if the GRand object has not been
91// seeded unpredictably.
92// In no other cases will any member function throw.
93// Whenever an exception is thrown, the Strong Guarantee holds, as
94// long as it holds for the type that originated the exception.
95class GRand {

149 // i
150 // Return uniformly distributed random integer in range [0, n-1], or
151 // 0 if n <= 0. Range is {0, 1} if no parameter given.
152 int i(int n=2)
153 { return (n <= 1) ? 0 : rand_integer(n); }
154
155 // d
156 // Return uniformly distributed random double in range [0.0, d) if
157 // d > 0.0, in range (d, 0.0] if d < 0.0, or 0.0 if d == 0.0. Range
158 // is [0.0, 1.0) if no parameter given.
159 double d(double d=1.0)
160 {
161 ck_seed();
162 if (d > 0.0)
163 return std::uniform_real_distribution<double>(0.0, d)(_rng);
164 if (d < 0.0)
165 return
166 -std::uniform_real_distribution<double>(0.0, -d)(_rng);
167 return 0.0;
168 }
169
170 // b
171 // Return random bool value. Probability of true is given value, or
172 // 0.5 if none given.
173 bool b(double p=0.5)
174 {
175 ck_seed();
176 if (p <= 0.0) return false;
177 if (p >= 1.0) return true;
178 return std::bernoulli_distribution(p)(_rng);
179 }

184 // An object of type GRand meets the requirements for Uniform Random
185 // Number Generator (C++11 26.5.1.3). An object of type GRand may be

229 // operator() - one argument
230 // Return random integer in range [0, n-1], or 0 if n <= 0. Result
231 // is uniformly distributed if n <= ULLONG_MAX.
232 // Requirements on Types:
233 // IntType must be copy constructible.
234 // IntType must be convertible to unsigned long long.
235 // unsigned long long must be convertible to IntType.
236 template <typename IntType>
237 IntType operator()(IntType n)
238 {
239 if (n <= IntType(1ULL)) return IntType(0ULL);
240 // "ULL" in above line because of our requirements on types.
241 typedef typename
242 std::common_type<IntType, unsigned long long>::type TT;
243 // Explicit type promotion avoids signed/unsigned warnings.
244 if (TT(n) <= TT(std::numeric_limits<unsigned long long>::max()))
245 // Above is correct for signed or unsigned n, since n > 0.
246 return IntType(rand_integer((unsigned long long)(n)));
247 else
248 // If n is too big for (unsigned long long), we do the best
249 // we can.
250 return IntType(rand_integer(
251 std::numeric_limits<unsigned long long>::max()));
252 }
253 // Simpler non-template versions of operator()(n) for the 8
254 // fundamental types that rand_integer can take.
255#define GRAND_OP_PAREN(T) T operator()(T n) \
256 { return (n <= (T)(1)) ? (T)(0) : rand_integer(n); }

273 // rand_integer
274 // Return uniformly distributed random integer in range [0, n-1].
275 // Pre:
276 // n >= 2.
277 // Requirements on Types:
278 // IntType must be short, int, long, long long, or the unsigned
279 // version of one of these (to avoid undefined behavior from
280 // std::uniform_int_distribution).
281 // This function is used in both i(n) and operator()(n). We avoid
282 // the n <= 1 check here (thus the precondition) so that we can
283 // handle negative n correctly.
284 template <typename IntType>
285 IntType rand_integer(IntType n)
286 {
287 ck_seed();
288#ifdef __GNUC__
289// g++ 4.7.* incorrectly gives conversion warnings when IntType is short
290#pragma GCC diagnostic push
291#pragma GCC diagnostic ignored "-Wconversion"
292#endif
293 return std::uniform_int_distribution<IntType>(
294 IntType(0),
295 n-IntType(1))
296 (_rng);
297#ifdef __GNUC__
298#pragma GCC diagnostic pop
299#endif
300 }
301
302 // ck_seed
303 // If _seed_needed is true, then do nondeterministic seed and set
304 // _seed_needed to false; otherwise do nothing.
305 //
306 // This function exists to allow a default-constructed GRand object
307 // to be given a specified seed later, without nondeterministic
308 // seeding being done in the meantime. Thus we avoid crashes in
309 // environments where nondeterministic seeding is not available
310 // (e.g., O. Lawlor's NetRun).
311 void ck_seed()
312 {
313 if (!_seed_needed)
314 return;
315 _rng.seed(rng_type::result_type(std::random_device()()));
316 _seed_needed = false;
317 }
318
